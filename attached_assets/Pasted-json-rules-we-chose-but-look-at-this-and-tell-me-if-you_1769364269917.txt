CREATE OR REPLACE FUNCTION archive_old_policy_versions()
RETURNS TRIGGER AS $$
BEGIN
    -- If we are activating a rule, archive other active rules with the same name/category
    IF NEW.status = 'active' THEN
        UPDATE policy_rules 
        SET status = 'archived', expiry_date = NOW()
        WHERE rule_name = NEW.rule_name 
          AND category = NEW.category 
          AND id != NEW.id 
          AND status = 'active';
    END IF;
    RETURN NEW;
END;

1. The Policy Runner Service
This service pulls the active rules from the SQL view we created and evaluates them against the agent's current context.

TypeScript
import { Engine } from 'json-rules-engine';
import { Pool } from 'pg'; // Assuming PostgreSQL

export class PolicyExecutionService {
  private pool: Pool;

  constructor(dbPool: Pool) {
    this.pool = dbPool;
  }

  async evaluatePolicy(category: string, facts: any) {
    const engine = new Engine();

    // 1. Fetch only the relevant, active rules from our DB View
    const { rows: rules } = await this.pool.query(
      'SELECT rule_definition FROM active_agent_rules WHERE category = $1',
      [category]
    );

    // 2. Load them into the engine
    rules.forEach(row => {
      engine.addRule(row.rule_definition);
    });

    // 3. Execute
    try {
      const { events, failureEvents } = await engine.run(facts);
      
      return {
        allowed: events.length === 0 || !events.some(e => e.params.severity === 'block'),
        decisions: events.map(e => e.params.message),
        auditTrail: events.map(e => e.type)
      };
    } catch (error) {
      console.error("Policy Engine Failure:", error);
      throw new Error("Could not verify policy compliance.");
    }
  }
}
