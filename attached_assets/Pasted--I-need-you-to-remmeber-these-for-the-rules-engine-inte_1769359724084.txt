  How It Works:                                                                                                                                     
                                                                                                                                                    
  1. Teams use Retool - Each team (FinOps, VRO, PMO, OCM, etc.) logs into their dedicated Rule Editor                                               
  2. Configure rules - Set thresholds, select actions, choose target agents

  4. Engine reads rules - AgentCollaborationRulesEngine.ts queries the database                                                                     
  5. Agents evaluate - Deep Agents check rules during execution                                                                                     
  6. Trigger actions - When conditions met, send A2A messages to target agents

  CREATE TABLE collaboration_rules (                                                                                                                
    id TEXT PRIMARY KEY,                                                                                                                            
    agent_id TEXT NOT NULL,           -- 'finops', 'vro', 'pmo', 'ocm', etc.                                                                        
    rule_name TEXT NOT NULL,                                                                                                                        
    description TEXT,

    priority TEXT DEFAULT 'medium',                                                                                                                 
    created_by TEXT,                                                                                                                                
    created_at TIMESTAMP DEFAULT NOW(),                                                                                                             
    updated_at TIMESTAMP DEFAULT NOW(),                                                                                                             
    version INTEGER DEFAULT 1,                                                                                                                      
    change_history JSONB DEFAULT '[]'                                                                                                               
  );
